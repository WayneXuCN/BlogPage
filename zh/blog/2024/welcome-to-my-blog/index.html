代码示例

import numpy as np
import networkx as nx

def calculate_resilience(G: nx.Graph, failure_nodes: list) -> float:
    """计算网络在节点失效后的韧性指标"""
    G_copy = G.copy()
    G_copy.remove_nodes_from(failure_nodes)

    if len(G_copy) == 0:
        return 0.0

    largest_cc = max(nx.connected_components(G_copy), key=len)
    return len(largest_cc) / len(G)
