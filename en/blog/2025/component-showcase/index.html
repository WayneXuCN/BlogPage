Basic Code Block

def fibonacci(n: int) -> int:
    """Calculate the nth Fibonacci number"""
    if n <= 1:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

# Test
for i in range(10):
    print(f"F({i}) = {fibonacci(i)}")

function createUser(name: string, email: string): User {

    createdAt: new Date()
